# To do timing of the full project:
python -m cProfile -o times.prof main_mf.py


# Then to retrieve a readable format:
python -m pstats times.prof

# and pass arguments

times.prof% strip
times.prof% sort cumtime
times.prof% stats 10


# all commands can be seen using:
times.prof% help

# werkt erg goed en visualiseert:
snakeviz times.prof
